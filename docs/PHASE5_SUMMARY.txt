handleInlineUpdate Function:
```tsx
const handleInlineUpdate = async (
  teacherId: string,
  field: 'maxLoad' | 'preps' | 'students',
  value: number
) => {
  try {
    await teacherApi.update(teacherId, { [field]: value });
    toast.success('Updated', `${field} updated successfully`);
    onTeacherUpdate?.();
  } catch (error: any) {
    toast.error('Update failed', error.message);
    throw error;
  }
};
```
